            measurementId: "G-LE2M9GDKCT"
        };

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);

        // ROUTER GUARD
        onAuthStateChanged(auth, (user) => {
            if (!user) {
                window.location.href = "index.html";
            }
        });

        // Local Chat Box Rendering Logic
        window.transmitLocalMessage = function(event) {
            event.preventDefault();
            const inputElement = document.getElementById('message-input-text');
            const streamBox = document.getElementById('messages-stream-box');
            
            if (inputElement.value.trim() === "") return;

            // Generate outgoing bubble
            const bubble = document.createElement('div');
            bubble.className = "msg-bubble outgoing";
            bubble.innerText = inputElement.value;
            
            streamBox.appendChild(bubble);
            inputElement.value = ""; // Clear Input
            
            // Auto scroll to bottom
            streamBox.scrollTop = streamBox.scrollHeight;
        }

        // Active state context switcher switcher 
        window.switchActiveChat = function(username, avatarUrl) {
            document.getElementById('active-chat-username').innerText = username;
            document.getElementById('active-chat-avatar').src = avatarUrl;
            
            // Remove active classes and switch highlight
            const items = document.querySelectorAll('.chat-user-item');
            items.forEach(i => i.classList.remove('active'));
            event.currentTarget.classList.add('active');
            
            alert(`Chat switched to ${username}! Connecting to firestore message collections soon.`);
        }

        // Logout Process routine
        window.logoutUser = function() {
            signOut(auth).then(() => {
                window.location.href = "index.html";
            });
        }
