        // Local file content rendering preview generator
        window.previewFile = function() {
            const file = document.getElementById('file-selector').files[0];
            const previewHolder = document.getElementById('media-preview-holder');
            const prompt = document.getElementById('upload-prompt');
            
            if (!file) return;

            prompt.style.display = 'none';
            previewHolder.innerHTML = ''; // Reset holder

            const reader = new FileReader();
            reader.onload = function(e) {
                if (file.type.startsWith('image/')) {
                    const img = document.createElement('img');
                    img.src = e.target.result;
                    previewHolder.appendChild(img);
                } else if (file.type.startsWith('video/')) {
                    const video = document.createElement('video');
                    video.src = e.target.result;
                    video.controls = true;
                    video.autoplay = true;
                    video.muted = true;
                    previewHolder.appendChild(video);
                }
            }
            reader.readAsDataURL(file);
        }

        // Action Trigger logic for Cloud submissions
