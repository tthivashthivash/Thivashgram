        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
        import { getAuth, onAuthStateChanged, updatePassword, deleteUser, signOut } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js";

        const firebaseConfig = {
            apiKey: "AIzaSyCR3xMj0rcdC1jRQyGePjtO9rpd8k5Om3E",
            authDomain: "thivashgram.firebaseapp.com",
            projectId: "thivashgram",
            storageBucket: "thivashgram.firebasestorage.app",
            messagingSenderId: "515740477126",
            appId: "1:515740477126:web:44605f2b80c50d86d799c7",
            measurementId: "G-LE2M9GDKCT"
        };

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        let currentUserObj = null;

        // ROUTER GUARD & USER LOADER
        onAuthStateChanged(auth, (user) => {
            if (!user) {
                window.location.href = "index.html";
            } else {
                currentUserObj = user;
                const userIdent = user.displayName ? user.displayName : user.email.split('@')[0];
                document.getElementById('settings-username-label').innerText = userIdent;
                
                if (user.photoURL) {
                    document.getElementById('settings-avatar-preview').src = user.photoURL;
                }
            }
        });

        // Firebase Password Update Handler
        window.handlePasswordUpdate = function(event) {
            event.preventDefault();
            const newPass = document.getElementById('new-password').value;

            if (currentUserObj) {
                updatePassword(currentUserObj, newPass)
                    .then(() => {
                        alert("Password updated successfully!");
                        document.getElementById('password-form').reset();
                    })
                    .catch((error) => {
                        alert("Error: " + error.message.replace("Firebase: ", ""));
                    });
            }
        }

        // DP Change Placeholder Logic
        window.uploadNewDP = function() {
            const file = document.getElementById('dp-file-selector').files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                document.getElementById('settings-avatar-preview').src = e.target.result;
                alert("UI Updated! Next, we will link Firebase Storage to save this DP permanently.");
            }
            reader.readAsDataURL(file);
        }

        // Firebase Account Deletion Handler
        window.handleAccountDeletion = function() {
            const confirmDelete = confirm("Are you absolutely sure you want to delete your ThivashGram account?");
            if (confirmDelete && currentUserObj) {
                deleteUser(currentUserObj)
                    .then(() => {
                        alert("Account deleted successfully.");
                        window.location.href = "index.html";
                    })
                    .catch((error) => {
                        alert("For security reasons, please log in again before deleting your account.");
                    });
            }
        }

        // Sign Out Core Module trigger
        window.logoutUser = function() {
            signOut(auth).then(() => {
                window.location.href = "index.html";
            });
        }
    </script>
